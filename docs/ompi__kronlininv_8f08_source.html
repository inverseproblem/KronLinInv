142  integer,intent(in) :: nit,nunits,startpoint
143  integer,allocatable,intent(out) :: scheduling(:),looping(:,:)
144  integer :: i,nitcpu,resto
145  if (nit<=nunits) then
146  write(*,*) myrank,": spredwork(): nit<=nunits"

149  end if

164  looping(i,2) = sum(scheduling(1:i)) + startpoint - 1

563  subroutine blockpostcov(U1,U2,U3, diaginvlambda, &
564  iUCm1,iUCm2,iUCm3, astart,aend,bstart,bend, postC)
565  !!
566  !! Calculates a block of the posterior covariance
567  !!
568  real(dp),intent(in) :: U1(:,:),U2(:,:),U3(:,:),iUCm1(:,:), &
569  iUCm2(:,:),iUCm3(:,:)
570  !! diaginvlambda = (I + lam1 x lam2 x lam3 )^-1
571  real(dp),intent(in) :: diaginvlambda(:) !! diagonal/vector central factor
572  real(dp),intent(out) :: postC(:,:)
573  integer,intent(in) :: astart,aend,bstart,bend
574 
575  integer :: a,b,irow,i,j,sender
576  real(dp),allocatable :: row2(:),col1(:)
577  integer :: Nr12,Nc1
578  integer :: Ni,Nj,Nk,Nl,Nm,Nn,Na,Nb,ntota,ntotb
579 
580  integer,allocatable :: av(:)!,bv(:)
581  integer,allocatable :: iv(:),jv(:),kv(:) !,lv(:),mv(:),nv(:)
582  integer :: p,everynit,impicount,mytotit,rirow
583  real(dp) :: startt,firststartt,endt
584  integer,allocatable :: scheduling(:),looping(:,:)
585  real(dp),allocatable :: postcmpi(:,:),recvrow(:),sendrow(:)
586  character(len=30) :: loopinfo
587 
588  if (myrank==masterrank) then
589  write(output_unit,*) "posteriormean(): calculating posterior mean... [using OpenMPI]"
590  end if
591 
592  ni = size(u1,1)
593  nl = size(u1,2)
594  nj = size(u2,1)
595  nm = size(u2,2)
596  nk = size(u3,1)
597  nn = size(u3,2)
598  na = ni*nj*nk
599  nb = nl*nm*nn
600 
601  !! check shape of output array
602  if (na /= nb) then
603  write(*,*) '(Na /= Nb)', na,nb
604  stop
605  end if
606  ntota = aend-astart+1
607  ntotb = bend-bstart+1
608  if ( (size(postc,1)/=ntota) .or. (size(postc,2)/=ntotb)) then
609  write(*,*) "Wrong size of the posterior covariance block array."
610  stop
611  end if
612  !! check limits of requested block
613  if ( (astart<1) .or. (aend>na) .or. (astart>aend) .or. &
614  (bstart<1) .or. (bend>na) .or. (bstart>bend) ) then
615  write(*,*) "Wrong size of the requested block array."
616  stop
617  end if
618 
619  !! vectorize row and col calculations for Kron prod AxBxC
620  allocate(av(na),iv(na),jv(na),kv(na))
621  !allocate(bv(Nb),lv(Nb),mv(Nb),nv(Nb))
622  forall(p = 1:na) av(p) = p
623  !forall(p = 1:Nb) bv(p) = p
624 
625  !! vectors containing all possible indices for
626  !! row calculations of Kron prod AxBxC
627  iv = (av-1)/(nk*nj)+1
628  jv = (av-1-(iv-1)*nk*nj)/nk + 1
629  kv = av-(jv-1)*nk-(iv-1)*nk*nj
630 
631  !! allocate stuff
632  nr12 = size(u1,2)* size(u2,2)* size(u3,2)
633  nc1 = size(iucm1,1)* size(iucm2,1)* size(iucm3,1)
634  allocate(row2(nr12),col1(nc1))
635 
636  !!------------------------
637  if (ntota<1000) then
638  everynit = 20
639  else
640  everynit = ntota/1000
641  end if
642 
643  !!---------------------------------
644  firststartt = mpi_wtime()
645  loopinfo = "blockpostcov() "
646  if (myrank==masterrank) write(output_unit,*)
647  !! calculate scheduling for a
648  call spreadwork(ntota,numcpus,scheduling,looping,1)
649  allocate(postcmpi(scheduling(myrank+1),ntotb),recvrow(ntotb),sendrow(ntotb))
650  mytotit = scheduling(myrank+1)
651 
652  !!print*,myrank,":",looping(myrank+1,1),"|",looping(myrank+1,2),"|",scheduling
653 
654  !!-----------------------
655  startt = mpi_wtime()
656  impicount = 0
657  do irow=looping(myrank+1,1),looping(myrank+1,2)
658  !! set a
659  a=irow+astart-1
660  !! info
661  if ( (myrank==masterrank) .and. (mod(irow,everynit)==0) ) then
662  call timeinfo(mytotit,irow,startt,loopinfo)
663  end if
664 
665  !! row first two factors
666  !! a row x diag matrix
667  !!row2 = U1(iv(a),lv) * U2(jv(a),mv) * U3(kv(a),nv) * diaginvlambda
668  row2 = u1(iv(a),iv) * u2(jv(a),jv) * u3(kv(a),kv) * diaginvlambda
669 
670  j=0
671  do b=bstart,bend
672  j=j+1
673  !! calculate one row of first TWO factors
674  !!call columnAxBxC(Ni,Nj,Nk,Nl,Nm,Nn, iUCm1,iUCm2,iUCm3,b,col1)
675  col1 = iucm1(iv,iv(b)) * iucm2(jv,jv(b)) * iucm3(kv,kv(b))
676 
677  !! calculate one element of the posterior covariance
678  i = irow-looping(myrank+1,1)+1
679  postcmpi(i,j) = sum(row2*col1)
680 
681  end do
682 
683  end do
684 
685  !! collect all results in masterrank
686  call para_barrier()
687  do sender=0,numcpus-1
688  if ( (myrank==sender) .or. (myrank==masterrank) ) then
689  do i=looping(sender+1,1),looping(sender+1,2)
690  !!----------------------
691  !! send/rec each row
692  !print*,"==>",myrank,": sender=",sender,"i=",i
693  if (myrank==sender) sendrow = postcmpi(i,:)
694  call para_srarr1ddp(sender,masterrank,i,sendrow,recvrow)
695  !call para_srint(myrank,masterrank,impicount,irow,rirow)
696  if (myrank==masterrank) postc(i,:) = recvrow
697  end do
698  end if
699  end do
700 
701  call para_barrier()
702  if (myrank==masterrank) then
703  write(output_unit,*)
704  endt = MPI_Wtime()
705  print*,"blockpostcov():",endt-firststartt," MPI wall clock time"
706  end if
707 
708  end subroutine blockpostcov
